\subsection{算法}

如\cref{alg:冒泡排序算法}。

\begin{algorithm}[htb]
\caption{冒泡排序算法}
\label{alg:冒泡排序算法}
\KwIn{无序数组 $A[1 \dots n]$}
\KwOut{排序后的数组 $A$}
\For{$i = 1$ \KwTo $n-1$}{
    \For{$j = 1$ \KwTo $n-i$}{
        \If{$A[j] > A[j+1]$}{
            交换 $A[j]$ 和 $A[j+1]$
        }
    }
}
\end{algorithm}
